### Loading Sprites
```kotlin
// Convert pixel art string to ImageBitmap
fun PixelArt.toImageBitmap(scale: Int = 2): ImageBitmap {
    val width = rows[0].length * scale
    val height = rows.size * scale
    val bitmap = ImageBitmap(width, height)

    Canvas(bitmap).apply {
        rows.forEachIndexed { y, row ->
            row.forEachIndexed { x, char ->
                val color = palette[char] ?: Color.Transparent
                drawRect(
                    color = color,
                    topLeft = Offset(x * scale.toFloat(), y * scale.toFloat()),
                    size = Size(scale.toFloat(), scale.toFloat())
                )
            }
        }
    }

    return bitmap
}
```

### Animation System
```kotlin
data class AnimatedSprite(
    val frames: List<PixelArt>,
    val frameDuration: Long = 200L // milliseconds
)

@Composable
fun AnimatedSpriteRenderer(sprite: AnimatedSprite) {
    var currentFrame by remember { mutableStateOf(0) }

    LaunchedEffect(Unit) {
        while (true) {
            delay(sprite.frameDuration)
            currentFrame = (currentFrame + 1) % sprite.frames.size
        }
    }

    SpriteRenderer(sprite.frames[currentFrame])
}
```
